
# Events for Swedish claims on other countries

add_namespace = deluge_sweden

# Finland

news_event = {
	id = deluge_sweden.1
	title = deluge_sweden1title
	desc = deluge_sweden1desc
	
	picture = GFX_news_event_012
	
	major = yes
	
	is_triggered_only = yes
	
	option = {
		name = deluge_sweden1opta
		trigger = {
			OR = {
				original_tag = FIN
				original_tag = SOV
			}
		}
	}
	option = {
		name = deluge_sweden1optb
		trigger = {
			OR = {
				original_tag = SWE
				original_tag = SWD
			}
		}
	}
	option = {
		name = deluge_sweden1optc
		trigger = {
			NOT = {
				OR = {
					original_tag = FIN
					original_tag = SOV
					original_tag = SWE
					original_tag = SWD
				}
			}
		}
	}
}
