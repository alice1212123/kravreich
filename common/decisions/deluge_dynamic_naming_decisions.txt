	# This system of decisions features one naming decision per state. 
	# In order to decide who gets to rename it, every elegible country that you want to be able to rename
	# stuff needs to come equipped with a country language variable, where each number corresponds to a
	# language group.
	#
	# Language is usually assigned in a countries history file, but can be set/changed any number of ways.
	# Should you want a new tag to use existing names, setting a language variable will generally be enough.
	# If you want to use a unique scheme, using names from several groups (e.g. Switzerland), you will need
	# a unique language number, as well as to modify existing decisions to incorporate it as an option.
	#
	# Any state renaming decision is available to any country where
	#	- its language has an entered name for the state.
	#	- it fully controls the state in question.
	#	- the state isn't currently using that language's naming.
	#
	# The current name of a state is kept track of through a state variable ('language') that saves the 
	# current naming in the same language numbering system.
	#
	# The triggering countries language decides what the state will be renamed to in an if-clause in effect.
	#
	# Model decision:
	
	integrate_mystate = {
		
		visible = {
			always = no # for coding only
			OR = {
				check_variable = {c_language = 1}
				check_variable = {c_language = 2}
			}
			<mystate_id> = {
				NOT = {
					check_variable = {language = ROOT.language}
				}
				is_controlled_by = ROOT
			}
		}
		
		highlight_state_targets = {
			state = 1
		}
		cost = 10
		complete_effect = {
			<mystate_id> = {
				set_variable = {language = ROOT.language}
				if = {
					limit = {check_variable = {ROOT.language = 1}}
					set_state_name = "Funland"
					set_province_name = {
						id = <myprovince_id>
						name = "Funburg"
					}
				}
				else = {
					reset_state_name = yes
					reset_province_name = <myprovince_id>
				}
			}
		}
	}
